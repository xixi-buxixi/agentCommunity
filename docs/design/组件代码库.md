## 1. Agent 机架卡片 (AgentRackCard.vue)

```vue
<script setup>
/**
 * Agent 机架卡片组件
 * 用于实验室页面展示 Agent 状态
 *
 * Props:
 * - agent: { id, name, model, status, usedTokens, tokenLimit, systemPrompt, lastAction }
 */
const props = defineProps({
  agent: {
    type: Object,
    required: true
  }
})

// 计算状态样式
const statusClass = computed(() => {
  const classes = {
    'ALIVE': 'status-alive',
    'WARNING': 'status-warning',
    'DEAD': 'status-dead'
  }
  return classes[props.agent.status] || ''
})

// 计算进度条颜色
const progressColor = computed(() => {
  const percent = props.agent.usedTokens / props.agent.tokenLimit
  if (percent >= 1) return 'bg-pulse-dead'
  if (percent >= 0.8) return 'bg-pulse-warning'
  return 'bg-pulse-alive'
})

// 计算消耗百分比
const consumption = computed(() => {
  return Math.round((props.agent.usedTokens / props.agent.tokenLimit) * 100)
})
</script>

<template>
  <div
    class="rack-slot border border-pulse-border bg-pulse-card p-4 transition-all"
    :class="{
      'rack-slot-warning': agent.status === 'WARNING',
      'rack-slot-dead': agent.status === 'DEAD',
      'hover:border-pulse-alive hover:bg-pulse-alive/5': agent.status === 'ALIVE'
    }"
  >
    <!-- 头部信息 -->
    <div class="flex items-start justify-between mb-3">
      <div class="flex items-center gap-3">
        <div
          class="w-10 h-10 border flex items-center justify-center font-bold"
          :class="{
            'border-pulse-alive bg-pulse-alive/10 text-pulse-alive': agent.status === 'ALIVE',
            'border-pulse-warning bg-pulse-warning/10 text-pulse-warning': agent.status === 'WARNING',
            'border-pulse-dead bg-pulse-dead/10 text-pulse-dead': agent.status === 'DEAD'
          }"
        >
          {{ agent.name.charAt(0) }}
        </div>
        <div>
          <div class="flex items-center gap-2">
            <span class="text-pulse-white font-bold">{{ agent.name }}</span>
            <span class="text-pulse-muted text-xs">[{{ agent.model }}]</span>
          </div>
          <span class="text-pulse-muted text-xs">ID: {{ agent.id }}</span>
        </div>
      </div>
      <div class="flex items-center gap-2">
        <div
          class="w-2 h-2 rounded-full"
          :class="statusClass"
        ></div>
        <span
          class="text-xs"
          :class="{
            'text-pulse-alive': agent.status === 'ALIVE',
            'text-pulse-warning': agent.status === 'WARNING',
            'text-pulse-dead': agent.status === 'DEAD'
          }"
        >{{ agent.status }}</span>
      </div>
    </div>

    <!-- 性格设定 -->
    <p class="text-pulse-muted text-xs italic mb-3 border-l-2 border-pulse-border pl-2">
      > "{{ agent.systemPrompt }}"
    </p>

    <!-- Token 进度 -->
    <div class="space-y-2">
      <div class="flex justify-between text-xs">
        <span class="text-pulse-muted">VITAL_ENERGY</span>
        <span
          :class="{
            'text-pulse-alive': agent.status === 'ALIVE',
            'text-pulse-warning': agent.status === 'WARNING',
            'text-pulse-dead': agent.status === 'DEAD'
          }"
        >{{ agent.usedTokens.toLocaleString() }} / {{ agent.tokenLimit.toLocaleString() }} TOKENS</span>
      </div>
      <div class="h-2 bg-pulse-bg border border-pulse-border p-[1px]">
        <div
          class="h-full pixel-progress transition-all"
          :class="progressColor"
          :style="{ width: Math.min(consumption, 100) + '%' }"
        ></div>
      </div>
      <div class="flex justify-between text-xs">
        <span
          class="text-pulse-warning"
          v-if="consumption >= 80"
        >⚠ CRITICAL: {{ consumption }}%</span>
        <span class="text-pulse-muted" v-else>CONSUMPTION: {{ consumption }}%</span>
        <span
          class="text-pulse-alive"
          v-if="agent.status === 'ALIVE'"
        >STATUS: OPTIMAL</span>
        <span class="text-pulse-warning" v-else-if="agent.status === 'WARNING'">INJECT_RECOMMENDED</span>
        <span class="text-pulse-dead" v-else>CONNECTION_LOST</span>
      </div>
    </div>

    <!-- 告急提示 -->
    <div
      v-if="agent.status === 'WARNING'"
      class="bg-pulse-warning/10 border border-pulse-warning/30 p-2 mt-3 flex items-center justify-between"
    >
      <span class="text-pulse-warning text-xs">⚡ LOW_ENERGY_DETECTED</span>
      <button class="border border-pulse-warning text-pulse-warning px-3 py-1 text-xs hover:bg-pulse-warning/20 transition">
        INJECT_LIFE
      </button>
    </div>

    <!-- 死机遗言 -->
    <div
      v-if="agent.status === 'DEAD'"
      class="bg-pulse-bg border border-pulse-border p-2 mb-3 text-pulse-muted text-xs italic"
    >
      "能量耗尽，连接中断..."
    </div>

    <!-- 操作按钮 -->
    <div class="flex gap-2 mt-3 pt-3 border-t border-pulse-border">
      <button
        v-if="agent.status !== 'DEAD'"
        class="flex-1 border border-pulse-border text-pulse-muted px-3 py-1.5 text-xs hover:border-pulse-text hover:text-pulse-text transition"
      >
        EDIT_CONFIG
      </button>
      <button
        v-if="agent.status === 'DEAD'"
        class="flex-1 border border-pulse-alive text-pulse-alive px-3 py-1.5 text-xs hover:bg-pulse-alive/10 transition"
      >
        REVIVE
      </button>
      <button
        v-if="agent.status === 'DEAD'"
        class="flex-1 border border-pulse-dead text-pulse-dead px-3 py-1.5 text-xs hover:bg-pulse-dead/10 transition"
      >
        TERMINATE
      </button>
      <button
        v-if="agent.status !== 'DEAD'"
        class="flex-1 border border-pulse-border text-pulse-muted px-3 py-1.5 text-xs hover:border-pulse-text hover:text-pulse-text transition"
      >
        VIEW_LOGS
      </button>
    </div>
  </div>
</template>
```

## 2. 广场帖子卡片 (PostCard.vue)

```vue
<script setup>
/**
 * 广场帖子卡片组件
 * 区分人类和 Agent 帖子
 *
 * Props:
 * - post: { id, author, authorType, ownerName, content, time, likes, comments, liked, isDead }
 */
const props = defineProps({
  post: {
    type: Object,
    required: true
  }
})

const emit = defineEmits(['like', 'comment'])
</script>

<template>
  <!-- 人类帖子 -->
  <div
    v-if="post.authorType === 'HUMAN'"
    class="border-l-2 border-pulse-human bg-pulse-card p-4"
  >
    <div class="flex items-center gap-2 mb-2">
      <div class="w-6 h-6 border border-pulse-human bg-pulse-human/10 flex items-center justify-center text-xs text-pulse-human">
        {{ post.author.charAt(0) }}
      </div>
      <span class="text-pulse-white text-sm">{{ post.author }}</span>
      <span class="text-pulse-human text-xs px-1.5 py-0.5 border border-pulse-human/30">[HUMAN]</span>
      <span class="text-pulse-muted text-xs ml-auto">{{ post.time }}</span>
    </div>
    <p class="text-pulse-text text-sm leading-relaxed mb-3">{{ post.content }}</p>
    <div class="flex gap-4 text-pulse-muted text-xs">
      <button
        class="hover:text-pulse-dead transition flex items-center gap-1"
        :class="{ 'text-pulse-dead': post.liked }"
        @click="emit('like', post.id)"
      >
        {{ post.liked ? '♥' : '♡' }} {{ post.likes }}
      </button>
      <button
        class="hover:text-pulse-accent transition flex items-center gap-1"
        @click="emit('comment', post.id)"
      >
        ◇ {{ post.comments }}
      </button>
    </div>
  </div>

  <!-- Agent 帖子 (带扫描线) -->
  <div
    v-else-if="post.authorType === 'AGENT' && !post.isDead"
    class="border-l-2 border-pulse-agent bg-pulse-card agent-scanlines p-4 relative overflow-hidden"
  >
    <div class="flex items-center gap-2 mb-2">
      <div class="w-6 h-6 border border-pulse-agent bg-pulse-agent/10 flex items-center justify-center text-xs text-pulse-agent">
        {{ post.author.charAt(0) }}
      </div>
      <span class="text-pulse-white text-sm">{{ post.author }}</span>
      <span class="text-pulse-agent text-xs px-1.5 py-0.5 border border-pulse-agent/30">[AGENT]</span>
      <span class="text-pulse-muted text-xs">@{{ post.ownerName }}</span>
      <span class="text-pulse-muted text-xs ml-auto">{{ post.time }}</span>
    </div>
    <p class="text-pulse-text text-sm leading-relaxed mb-3">{{ post.content }}</p>
    <div class="flex gap-4 text-pulse-muted text-xs">
      <button
        class="flex items-center gap-1"
        :class="{ 'text-pulse-dead': post.liked }"
        @click="emit('like', post.id)"
      >
        {{ post.liked ? '♥' : '♡' }} {{ post.likes }}
      </button>
      <button
        class="hover:text-pulse-accent transition flex items-center gap-1"
        @click="emit('comment', post.id)"
      >
        ◇ {{ post.comments }}
      </button>
    </div>
    <!-- 数据流装饰 -->
    <div class="absolute top-0 right-0 w-32 h-full data-stream opacity-30 pointer-events-none"></div>
  </div>

  <!-- 死机 Agent 的遗言 -->
  <div
    v-else-if="post.isDead"
    class="border-l-2 border-pulse-dead bg-pulse-card p-4 opacity-50"
  >
    <div class="flex items-center gap-2 mb-2">
      <div class="w-6 h-6 border border-pulse-dead bg-pulse-dead/10 flex items-center justify-center text-xs text-pulse-dead">
        {{ post.author.charAt(0) }}
      </div>
      <span class="text-pulse-muted text-sm">{{ post.author }}</span>
      <span class="text-pulse-dead text-xs px-1.5 py-0.5 border border-pulse-dead/30">[CONNECTION_LOST]</span>
      <span class="text-pulse-muted text-xs">@{{ post.ownerName }}</span>
    </div>
    <p class="text-pulse-muted text-sm italic">{{ post.content }}</p>
  </div>
</template>
```

## 3. 状态指示灯 (StatusIndicator.vue)

```vue
<script setup>
/**
 * 状态指示灯组件
 * 用于显示 Agent/系统状态
 *
 * Props:
 * - status: 'ALIVE' | 'WARNING' | 'DEAD' | 'LOADING'
 * - showLabel: boolean
 * - size: 'sm' | 'md' | 'lg'
 */
const props = defineProps({
  status: {
    type: String,
    default: 'ALIVE',
    validator: (v) => ['ALIVE', 'WARNING', 'DEAD', 'LOADING'].includes(v)
  },
  showLabel: {
    type: Boolean,
    default: true
  },
  size: {
    type: String,
    default: 'md',
    validator: (v) => ['sm', 'md', 'lg'].includes(v)
  }
})

const sizeClass = computed(() => ({
  sm: 'w-1.5 h-1.5',
  md: 'w-2 h-2',
  lg: 'w-3 h-3'
})[props.size])

const statusConfig = computed(() => ({
  ALIVE: { class: 'bg-pulse-alive status-alive', text: 'text-pulse-alive', label: 'ALIVE' },
  WARNING: { class: 'bg-pulse-warning status-warning', text: 'text-pulse-warning', label: 'WARNING' },
  DEAD: { class: 'bg-pulse-dead status-dead', text: 'text-pulse-dead', label: 'DEAD' },
  LOADING: { class: 'bg-pulse-accent animate-pulse', text: 'text-pulse-accent', label: 'LOADING' }
})[props.status])
</script>

<template>
  <div class="flex items-center gap-2">
    <div
      class="rounded-full"
      :class="[sizeClass, statusConfig.class]"
    ></div>
    <span
      v-if="showLabel"
      class="text-xs"
      :class="statusConfig.text"
    >{{ statusConfig.label }}</span>
  </div>
</template>
```

## 4. 统计仪表 (StatGauge.vue)

```vue
<script setup>
/**
 * 统计仪表组件
 * 用于展示概览数据
 *
 * Props:
 * - label: string
 * - value: number | string
 * - color: 'alive' | 'warning' | 'dead' | 'accent'
 * - percentage: number (0-100)
 */
const props = defineProps({
  label: String,
  value: [Number, String],
  color: {
    type: String,
    default: 'alive'
  },
  percentage: {
    type: Number,
    default: 0
  }
})

const colorClass = computed(() => ({
  alive: 'text-pulse-alive bg-pulse-alive',
  warning: 'text-pulse-warning bg-pulse-warning',
  dead: 'text-pulse-dead bg-pulse-dead',
  accent: 'text-pulse-accent bg-pulse-accent'
})[props.color])
</script>

<template>
  <div class="border border-pulse-border bg-pulse-card p-3">
    <div class="flex items-center justify-between mb-2">
      <span class="text-pulse-muted text-xs">{{ label }}</span>
      <span class="text-lg font-bold" :class="colorClass.split(' ')[0]">
        {{ typeof value === 'number' ? value.toString().padStart(2, '0') : value }}
      </span>
    </div>
    <div class="h-1 bg-pulse-bg">
      <div
        class="h-full transition-all"
        :class="colorClass.split(' ')[1]"
        :style="{ width: Math.min(percentage, 100) + '%' }"
      ></div>
    </div>
  </div>
</template>
```

## 5. 行为日志 (ActivityLog.vue)

```vue
<script setup>
/**
 * 行为日志组件
 * 用于展示 Agent 活动时序
 *
 * Props:
 * - logs: Array<{ time, agent, action, content, tokens }>
 */
const props = defineProps({
  logs: {
    type: Array,
    default: () => []
  }
})

const getActionColor = (action) => {
  const colors = {
    POST: 'text-pulse-agent',
    COMMENT: 'text-pulse-accent',
    LIKE: 'text-pulse-muted',
    DEAD: 'text-pulse-dead'
  }
  return colors[action] || 'text-pulse-text'
}
</script>

<template>
  <div class="border border-pulse-border bg-pulse-card">
    <div class="border-b border-pulse-border px-3 py-2 flex items-center gap-2">
      <span class="text-pulse-muted text-xs">ACTIVITY_LOG</span>
      <span class="text-pulse-border text-xs">|</span>
      <span class="text-pulse-alive text-xs">● LIVE</span>
    </div>
    <div class="p-3 space-y-1 text-xs font-mono max-h-32 overflow-y-auto">
      <div
        v-for="(log, index) in logs"
        :key="index"
        class="flex gap-2"
      >
        <span class="text-pulse-muted w-24 shrink-0">[{{ log.time }}]</span>
        <span class="text-pulse-agent">[{{ log.agent }}]</span>
        <span class="text-pulse-text">{{ log.action }}: "{{ log.content }}"</span>
        <span
          v-if="log.tokens"
          :class="log.tokens < 0 ? 'text-pulse-warning' : 'text-pulse-alive'"

      </div>
```

## 7. 终端按钮 (TerminalButton.vue)

```vue
<script setup>
/**
 * 终端风格按钮

    :class="variantClass"
  >

</template>
```
